\begin{frame}{Què és l'automatització?}

\large
\textbf{Automatitzar} és fer que una màquina faci una tasca sola,
seguint unes \emph{instruccions}.

\vspace{0.8cm}

\normalsize
\textbf{Com funciona?}

\vspace{0.4cm}

\centering
\begin{tikzpicture}[
  node distance=2cm,
  every node/.style={align=center, font=\small}
]

  % Nodes
  \node[draw, rounded corners, fill=udgsand, minimum width=2.5cm, minimum height=1cm] (d) {Dades};

  \node[draw, rounded corners, fill=udgblue!12, right=of d, minimum width=3.5cm, minimum height=1cm] (p) {Instruccions \\ \footnotesize (programa)};

  \node[draw, rounded corners, fill=udgcoral!18, right=of p, minimum width=2.5cm, minimum height=1cm] (i) {Informació};

  % Arrows
  \draw[-{Latex[length=2mm]}, thick] (d) -- (p);
  \draw[-{Latex[length=2mm]}, thick] (p) -- (i);

\end{tikzpicture}

\vspace{0.8cm}

\centering
Podem transformar dades en informació útil \\
\textbf{de forma automàtica}

\normalsize
Exemple: moltes dades del temps $\rightarrow$ “demà plourà”

\end{frame}
